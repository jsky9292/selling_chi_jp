■ 자동 계산 공식

=(A2*B2)*(E2/100)                     // 구매대행 수수료
=SUM(F2:I2)                          // 고객 총 지불금액
=I2-(A2*B2+G2)                       // 순수익

월 고정비: [L2]
목표 순이익: [M2]
필요 주문수: =ROUNDUP((L2+M2)/J2,0)

import requests
from bs4 import BeautifulSoup
import time
import smtplib
from email.mime.text import MIMEText
import json

class JapanPriceMonitor:
    def __init__(self):
        self.target_products = [
            {
                'name': '상품명',
                'url': 'https://example.co.jp/product/123',
                'target_price': 5000,  # 목표가격 (엔)
                'css_selector': '.price'
            }
        ]
        
    def get_price(self, url, selector):
        """웹사이트에서 가격 정보 추출"""
        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
        
        try:
            response = requests.get(url, headers=headers)
            soup = BeautifulSoup(response.content, 'html.parser')
            
            price_element = soup.select_one(selector)
            if price_element:
                # 일본 가격 텍스트에서 숫자 추출
                price_text = price_element.text.strip()
                price = int(''.join(filter(str.isdigit, price_text)))
                return price
            else:
                return None
                
        except Exception as e:
            print(f"가격 조회 오류: {e}")
            return None
    
    def send_alert(self, product_name, current_price, target_price):
        """목표가격 달성 시 알림 발송"""
        sender_email = "your_email@gmail.com"
        sender_password = "your_app_password"
        receiver_email = "alert@yourdomain.com"
        
        subject = f"🚨 가격 알림: {product_name}"
        body = f"""
        상품명: {product_name}
        현재가격: ¥{current_price:,}
        목표가격: ¥{target_price:,}
        절약금액: ¥{target_price - current_price:,}
        
        지금 바로 구매하세요!
        """
        
        msg = MIMEText(body)
        msg['Subject'] = subject
        msg['From'] = sender_email
        msg['To'] = receiver_email
        
        try:
            server = smtplib.SMTP('smtp.gmail.com', 587)
            server.starttls()
            server.login(sender_email, sender_password)
            server.send_message(msg)
            server.quit()
            print(f"알림 발송 완료: {product_name}")
        except Exception as e:
            print(f"알림 발송 실패: {e}")
    
    def monitor_prices(self):
        """가격 모니터링 실행"""
        while True:
            for product in self.target_products:
                print(f"체크 중: {product['name']}")
                
                current_price = self.get_price(
                    product['url'], 
                    product['css_selector']
                )
                
                if current_price:
                    print(f"현재가격: ¥{current_price:,}")
                    
                    if current_price <= product['target_price']:
                        self.send_alert(
                            product['name'],
                            current_price,
                            product['target_price']
                        )
                
                time.sleep(10)  # 10초 대기
            
            time.sleep(3600)  # 1시간 주기로 체크

# 실행
if __name__ == "__main__":
    monitor = JapanPriceMonitor()
    monitor.monitor_prices()

import requests
import time

class ExchangeRateMonitor:
    def __init__(self):
        self.api_key = "YOUR_API_KEY"  # exchangerate-api.com
        self.base_url = "https://v6.exchangerate-api.com/v6"
        self.target_rates = {
            'JPY': {'target': 9.5, 'direction': 'below'},  # 엔화 9.5원 이하
            'CNY': {'target': 185, 'direction': 'below'}    # 위안 185원 이하
        }
        
    def get_exchange_rates(self):
        """실시간 환율 조회"""
        url = f"{self.base_url}/{self.api_key}/latest/KRW"
        
        try:
            response = requests.get(url)
            data = response.json()
            
            if data['result'] == 'success':
                # KRW 기준이므로 역수 계산
                rates = {}
                rates['JPY'] = 1 / data['conversion_rates']['JPY']  # 1엔당 원화
                rates['CNY'] = 1 / data['conversion_rates']['CNY']  # 1위안당 원화
                
                return rates
            else:
                print("환율 조회 실패")
                return None
                
        except Exception as e:
            print(f"환율 조회 오류: {e}")
            return None
    
    def check_targets(self, current_rates):
        """목표 환율 달성 체크"""
        alerts = []
        
        for currency, current_rate in current_rates.items():
            if currency in self.target_rates:
                target_info = self.target_rates[currency]
                target_rate = target_info['target']
                direction = target_info['direction']
                
                if direction == 'below' and current_rate <= target_rate:
                    alerts.append({
                        'currency': currency,
                        'current': current_rate,
                        'target': target_rate,
                        'message': f"{currency} 환율이 목표치에 도달했습니다!"
                    })
                elif direction == 'above' and current_rate >= target_rate:
                    alerts.append({
                        'currency': currency,
                        'current': current_rate,
                        'target': target_rate,
                        'message': f"{currency} 환율이 목표치를 초과했습니다!"
                    })
        
        return alerts

    def log_rates(self, rates):

        log_data = {
            'timestamp': datetime.now().isoformat(),
            'rates': rates
        }
        
        # CSV 파일로 저장
        with open('exchange_rates.json', 'a') as f:

    def send_promotion_message(self, customer_segments):
        """세그먼트별 프로모션 메시지"""
        messages = {

    def _send_message(self, template_object, receiver_id=None):
        """실제 메시지 발송"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/x-www-form-urlencoded"
        }
        
        data = {
            "template_object": json.dumps(template_object)
        }
        
        if receiver_id:
            data["receiver_uuids"] = json.dumps([receiver_id])
        
        try:
            response = requests.post(self.base_url, headers=headers, data=data)
            return response.json()
        except Exception as e:
            print(f"메시지 발송 오류: {e}")
            return None

-- 구매대행 고객 RFM 분석 쿼리
WITH customer_metrics AS (
  SELECT 
    customer_id,
    customer_name,
    -- Recency: 최근 구매일로부터 경과 일수
    DATE_DIFF(CURRENT_DATE(), MAX(order_date), DAY) as recency_days,
    
    -- Frequency: 총 구매 횟수
    COUNT(DISTINCT order_id) as frequency,
    
    -- Monetary: 총 구매 금액
    SUM(order_amount) as monetary,
    
    -- 추가 메트릭
    AVG(order_amount) as avg_order_value,
    MIN(order_date) as first_purchase_date,
    MAX(order_date) as last_purchase_date
    
  FROM orders
  WHERE order_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 12 MONTH)
  GROUP BY customer_id, customer_name
),

rfm_scores AS (
  SELECT *,
    -- R Score (낮을수록 좋음)
    CASE 
      WHEN recency_days <= 30 THEN 5
      WHEN recency_days <= 90 THEN 4
      WHEN recency_days <= 180 THEN 3
      WHEN recency_days <= 365 THEN 2
      ELSE 1
    END as r_score,
    
    -- F Score (높을수록 좋음)
    CASE 
      WHEN frequency >= 20 THEN 5
      WHEN frequency >= 10 THEN 4
      WHEN frequency >= 5 THEN 3
      WHEN frequency >= 3 THEN 2
      ELSE 1
    END as f_score,
    
    -- M Score (높을수록 좋음)
    CASE 
      WHEN monetary >= 5000000 THEN 5  -- 500만원 이상
      WHEN monetary >= 2000000 THEN 4  -- 200만원 이상
      WHEN monetary >= 1000000 THEN 3  -- 100만원 이상
      WHEN monetary >= 500000 THEN 2   -- 50만원 이상
      ELSE 1
    END as m_score
    
  FROM customer_metrics
),

customer_segments AS (
  SELECT *,
    CONCAT(r_score, f_score, m_score) as rfm_code,
    
    -- 세그먼트 분류
    CASE 
      WHEN r_score >= 4 AND f_score >= 4 AND m_score >= 4 THEN 'Champions'
      WHEN r_score >= 3 AND f_score >= 3 AND m_score >= 3 THEN 'Loyal Customers'
      WHEN r_score >= 4 AND f_score <= 2 THEN 'New Customers'
      WHEN r_score <= 2 AND f_score >= 3 THEN 'At Risk'
      WHEN r_score <= 2 AND f_score <= 2 THEN 'Lost Customers'
      WHEN r_score >= 3 AND f_score <= 2 AND m_score >= 3 THEN 'Big Spenders'
      ELSE 'Others'
    END as segment,
    
    -- 추천 액션
    CASE 
      WHEN r_score >= 4 AND f_score >= 4 AND m_score >= 4 THEN '개인 맞춤 서비스 제공'
      WHEN r_score >= 3 AND f_score >= 3 AND m_score >= 3 THEN '로열티 프로그램 안내'
      WHEN r_score >= 4 AND f_score <= 2 THEN '온보딩 프로그램 진행'
      WHEN r_score <= 2 AND f_score >= 3 THEN '재구매 유도 이벤트'
      WHEN r_score <= 2 AND f_score <= 2 THEN '윈백 캠페인 진행'
      ELSE '기본 마케팅'
    END as recommended_action
    
  FROM rfm_scores
)

-- 최종 결과 출력
SELECT 
  segment,
  COUNT(*) as customer_count,
  ROUND(AVG(monetary), 0) as avg_clv,
  ROUND(AVG(frequency), 1) as avg_frequency,
  ROUND(AVG(recency_days), 0) as avg_recency
FROM customer_segments
GROUP BY segment
ORDER BY customer_count DESC;

-- 개별 고객 정보 (마케팅 액션용)
SELECT 
  customer_id,
  customer_name,
  segment,
  rfm_code,
  monetary,
  frequency,
  recency_days,
  recommended_action,
  -- 다음 구매 예상일
  DATE_ADD(last_purchase_date, 
    INTERVAL CAST(365/frequency AS INT64) DAY) as predicted_next_purchase
FROM customer_segments
ORDER BY monetary DESC, frequency DESC;
